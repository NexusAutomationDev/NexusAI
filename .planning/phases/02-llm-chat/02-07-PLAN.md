```tsx
/**
 * Chat index route — the right panel of the two-column layout.
 * Renders MessageList (history) + MessageInput (bottom-fixed) for the active conversation.
 *
 * D-34: Empty state when no conversation selected ("Comece uma conversa").
 * D-28: Cmd/Ctrl+K keyboard shortcut to create new conversation.
 * D-09: MessageInput is bottom-fixed (handled inside MessageInput via sticky bottom-0).
 */

import { useEffect, useCallback } from "react";
import { createFileRoute, useNavigate, useSearch } from "@tanstack/react-router";
import { MessageSquare } from "lucide-react";
import { z } from "zod";
import { useChatStore } from "@/lib/stores/chat";
import {
  useMessages,
  useDeleteMessage,
  useCreateConversation,
  useInsertUserMessage,
} from "@/lib/queries/chat";
import { MessageList } from "./components/MessageList";
import { MessageInput } from "./components/MessageInput";

// TanStack Router search params schema for conversationId
const searchSchema = z.object({
  conversationId: z.string().optional(),
});

export const Route = createFileRoute("/chat/")({
  validateSearch: searchSchema,
  component: ChatView,
});

function ChatView() {
  const navigate = useNavigate();
  const search = useSearch({ from: "/chat/" });
  const conversationId = search.conversationId ?? null;

  const { activeConversationId, setActiveConversationId } = useChatStore();
  const { mutateAsync: createConversation } = useCreateConversation();
  const { mutateAsync: deleteMessage } = useDeleteMessage();
  const { data: messages = [] } = useMessages(conversationId);

  // Sync conversationId from URL to store
  useEffect(() => {
    if (conversationId && conversationId !== activeConversationId) {
      setActiveConversationId(conversationId);
    }
  }, [conversationId, activeConversationId, setActiveConversationId]);

  // D-28: Cmd/Ctrl+K creates new conversation
  const handleNewConversation = useCallback(async () => {
    const conv = await createConversation();
    setActiveConversationId(conv.id);
    navigate({ to: "/chat/", search: { conversationId: conv.id } });
  }, [createConversation, setActiveConversationId, navigate]);

  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      // D-28: Cmd+K (macOS) or Ctrl+K (Windows/Linux)
      if ((e.metaKey || e.ctrlKey) && e.key === "k") {
        e.preventDefault();
        handleNewConversation();
      }
    };
    window.addEventListener("keydown", handleKeyDown);
    return () => window.removeEventListener("keydown", handleKeyDown);
  }, [handleNewConversation]);

  const handleDeleteMessage = async (messageId: string) => {
    if (!conversationId) return;
    await deleteMessage({ messageId, conversationId });
  };

  // D-34: Empty state — no conversation selected
  if (!conversationId) {
    return (
      <div className="flex h-full flex-col">
        {/* Empty state centered in available space */}
        <div className="flex flex-1 flex-col items-center justify-center gap-3 text-center px-8">
          <MessageSquare size={40} className="text-muted-foreground opacity-40" />
          <div>
            {/* UI-SPEC copywriting: "Comece uma conversa" */}
            <h2 className="text-lg font-medium text-foreground">
              Comece uma conversa
            </h2>
            <p className="mt-1 text-sm text-muted-foreground max-w-sm">
              {/* UI-SPEC copywriting: empty state body */}
              Digite sua pergunta ou ideia. O modelo escolhido responderá em tempo real.
            </p>
          </div>
        </div>
        {/* NOTE: No MessageInput in empty state — user clicks "Nova Conversa" in conversation list */}
      </div>
    );
  }

  // Active conversation view
  return (
    // D-09: flex column, MessageList takes flex-1, MessageInput sticks to bottom
    <div className="flex h-full flex-col overflow-hidden">
      {/* Message list — scrollable, fills available space */}
      <MessageList
        messages={messages}
        conversationId={conversationId}
        onDeleteMessage={handleDeleteMessage}
        // Edit: re-send edited message (D-24: edit = delete + re-send)
        onEditMessage={async (messageId, newContent) => {
          // Soft-delete original message, insert new version
          await deleteMessage({ messageId, conversationId });
          // The MessageInput's handleSend will be triggered by the caller
          // For Phase 2: just delete the message, let user re-type (or re-send is handled in MessageInput)
        }}
      />

      {/* D-09: Bottom-fixed message input */}
      <MessageInput
        conversationId={conversationId}
        onSendComplete={() => {
          // Optional: any post-send actions (scroll is handled in MessageList)
        }}
      />
    </div>
  );
}
```

2. Or use TanStack Router's built-in search validation without zod: `validateSearch: (search) => ({ conversationId: search.conversationId as string | undefined })`

Check existing Phase 1 route files to see if zod is used there. If it's not used, use the simpler approach.
