      attachments: [],

    expect(result).toHaveProperty('base64Data');
    expect(typeof result.base64Data).toBe('string');
    expect(result.base64Data.length).toBeGreaterThan(0);
  });

  it('FileAttachment base64Data is valid base64 string', async () => {
    const result = await invoke<{ base64Data: string }>('pick_and_encode_file');
    // Valid base64 only contains A-Z, a-z, 0-9, +, /, and = padding

  it('send_message accepts attachments array', async () => {
    const attachment = await invoke<{ filename: string; mimeType: string; base64Data: string }>('pick_and_encode_file');
    const result = await invoke<{ id: string }>('send_message', {
      conversationId: 'conv-1',
      content: 'Analyze this file',
      model: 'gpt-4o',
      attachments: [attachment],

  });
});
